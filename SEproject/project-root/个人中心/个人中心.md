    <script>
        // 获取用户ID
        const userId = localStorage.getItem("userid");

        // 加载用户信息和用户基本信息
        function loadUserData() {
            if (!userId) {
                alert("未登录，请先登录");
                window.location.href = "login2.html";
                return;
            }

            // 获取用户基本信息
            fetch(`http://localhost:8080/user/${userId}`)
                .then(response => response.json())
                .then(user => {
                    document.getElementById("usernameDisplay").textContent = user.username || "未设置";
                    document.getElementById("userIdDisplay").textContent = user.userid;
                });

            // 获取用户详细信息
            fetch(`http://localhost:8080/userinfo/${userId}`)
                .then(response => response.json())
                .then(userInfo => {
                    if (userInfo) {
                        // 填充表单
                        document.getElementById("edit-employeeid").value = userInfo.employeeid || "";
                        document.getElementById("edit-email").value = userInfo.email || "";
                        document.getElementById("edit-gender").value = userInfo.gender || 2;
                        document.getElementById("edit-age").value = userInfo.age || "";
                        document.getElementById("edit-phone").value = userInfo.phone || "";

                        // 更新显示
                        document.getElementById("employeeIdDisplay").textContent = userInfo.employeeid || "未设置";
                        document.getElementById("emailDisplay").textContent = userInfo.email || "未设置";
                        document.getElementById("genderDisplay").textContent =
                            userInfo.gender === 0 ? "男" : userInfo.gender === 1 ? "女" : "未知";
                        document.getElementById("ageDisplay").textContent = userInfo.age || "-";
                        document.getElementById("phoneDisplay").textContent = userInfo.phone || "未设置";

                        // 设置头像
                        if (userInfo.avatar) {
                            document.getElementById("userAvatar").src = userInfo.avatar;
                            document.getElementById("avatarPreview").src = userInfo.avatar;
                        }
                    }
                });
        }

        // 保存用户信息
        document.getElementById("userInfoForm").addEventListener("submit", function (e) {
            e.preventDefault();

            const userInfo = {
                userid: parseInt(userId),
                employeeid: document.getElementById("edit-employeeid").value,
                email: document.getElementById("edit-email").value,
                gender: parseInt(document.getElementById("edit-gender").value),
                age: parseInt(document.getElementById("edit-age").value) || 0,
                phone: document.getElementById("edit-phone").value
            };

            fetch("http://localhost:8080/userinfo", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify(userInfo)
            })
                .then(response => response.json())
                .then(data => {
                    alert("信息保存成功");
                    loadUserData(); // 重新加载数据
                })
                .catch(error => {
                    console.error("保存失败:", error);
                    alert("保存失败，请重试");
                });
        });

        // 头像上传预览功能
        document.getElementById('avatarInput').addEventListener('change', function (e) {
            const file = e.target.files[0];
            if (file) {
                // 检查文件大小
                if (file.size > 2 * 1024 * 1024) {
                    alert("头像大小不能超过2MB");
                    return;
                }

                const reader = new FileReader();
                reader.onload = function (event) {
                    document.getElementById('avatarPreview').src = event.target.result;
                    document.getElementById('userAvatar').src = event.target.result;

                    // 这里可以添加上传头像到服务器的代码
                    uploadAvatar(file);
                };
                reader.readAsDataURL(file);
            }
        });

        // 上传头像到服务器
        function uploadAvatar(file) {
            const formData = new FormData();
            formData.append("avatar", file);
            formData.append("userid", userId);

            fetch("http://localhost:8080/userinfo/avatar", {
                method: "POST",
                body: formData
            })
                .then(response => {
                    if (response.ok) {
                        console.log("头像上传成功");
                    } else {
                        console.error("头像上传失败");
                    }
                })
                .catch(error => {
                    console.error("头像上传出错:", error);
                });
        }

        // 修改密码功能
        document.getElementById("changePasswordBtn").addEventListener("click", function () {
            document.getElementById("passwordModal").style.display = "flex";
        });

        document.getElementById("cancelPasswordChange").addEventListener("click", function () {
            document.getElementById("passwordModal").style.display = "none";
        });

        document.getElementById("passwordForm").addEventListener("submit", function (e) {
            e.preventDefault();

            const oldPassword = document.getElementById("oldPassword").value;
            const newPassword = document.getElementById("newPassword").value;
            const confirmNewPassword = document.getElementById("confirmNewPassword").value;

            if (newPassword !== confirmNewPassword) {
                alert("两次输入的新密码不一致");
                return;
            }

            fetch("http://localhost:8080/user/change-password", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({
                    userId: parseInt(userId),
                    oldPassword: oldPassword,
                    newPassword: newPassword
                })
            })
                .then(response => {
                    if (response.ok) {
                        return response.text();
                    } else {
                        throw new Error("密码修改失败");
                    }
                })
                .then(message => {
                    alert(message);
                    document.getElementById("passwordModal").style.display = "none";
                    document.getElementById("passwordForm").reset();
                })
                .catch(error => {
                    alert(error.message);
                });
        });

        // 页面加载时获取用户数据
        window.addEventListener("load", loadUserData);
    </script>
